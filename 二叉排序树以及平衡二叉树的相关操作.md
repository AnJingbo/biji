class TreeNode{
    int val;
    TreeNode left;
    TreeNode right;

    public TreeNode(int val){
        this.val = val;
    }
    public String toString(){
        return "val = " + val;
    }

    public void add(TreeNode node){

        if(node.val <= this.val){
            if(this.left == null){
                this.left = node;
            }else{
                this.left.add(node);
            }
        }
        if(node.val > this.val){
            if(this.right == null){
                this.right = node;
            }else{
                this.right.add(node);
            }
        }
    }

    public void midTraverse(){

        if(this.left != null){
            this.left.midTraverse();
        }
        System.out.println(this);
        if(this.right != null){
            this.right.midTraverse();
        }
    }

    public TreeNode findTargetNode(int val){
        if(this.val == val){
            return this;
        }
        TreeNode res = null;
        if(this.left != null){
            res = this.left.findTargetNode(val);
        }
        if(res != null){
            return res;
        }
        if(this.right != null){
            res = this.right.findTargetNode(val);
        }
        return res;
    }

    public TreeNode findParentNode(int val){
        if((this.left != null && this.left.val == val) || (this.right != null && this.right.val == val)){
            return this;
        }
        TreeNode res = null;
        if(this.left != null){
            res = this.left.findParentNode(val);
        }
        if(res != null){
            return res;
        }
        if(this.right != null){
            res = this.right.findParentNode(val);
        }
        return res;
    }
    public void delNode(int val){
        TreeNode targetNode = findTargetNode(val);
        TreeNode parentNode = findParentNode(val);
        if(targetNode.left == null && targetNode.right == null) {
            if(parentNode.left == targetNode){
                parentNode.left = null;
            }
            if(parentNode.right == targetNode){
                parentNode.right = null;
            }
        }else if(targetNode.left != null && targetNode.right != null){
            TreeNode temp = targetNode.right;
            while(temp.left != null){
                temp = temp.left;
            }
            delNode(temp.val);
            targetNode.val = temp.val;
        }else{
            if(parentNode.left == targetNode){
                if(targetNode.left != null){
                    parentNode.left = targetNode.left;
                }else if(targetNode.right != null){
                    parentNode.left = targetNode.right;
                }
            }
            if(parentNode.right == targetNode){
                if(targetNode.left != null){
                    parentNode.right = targetNode.left;
                }else if(targetNode.right != null){
                    parentNode.right = targetNode.right;
                }
            }
        }
    }

    public int height(){
        return Math.max((this.right == null) ? 0 : this.right.height(), (this.left == null) ? 0 : this.left.height()) + 1;
    }

    public int leftHeight(){
        if(this.left == null){
            return 0;
        }
        return this.left.height();
    }
    public int rightHeight(){
        if(this.right == null){
            return 0;
        }
        return this.right.height();
    }

    public void rotate(){
        if(rightHeight() - leftHeight() > 1){
            TreeNode newNode = new TreeNode(this.val);
            newNode.left = this.left;
            newNode.right = this.right.left;
            this.val = this.right.val;
            this.right = this.right.right;
            this.left = newNode;
        }else {
            TreeNode newNode = new TreeNode(this.val);
            newNode.right = this.right;
            newNode.left = this.left.right;
            this.val = this.left.val;
            this.left = this.left.left;
            this.right = newNode;
        }
    }
}


class BinarySortTree{
    TreeNode root;

    public void add(TreeNode node){
        if(node == null){
            return;
        }
        if(root == null){
            root = node;
            return;
        }
        root.add(node);
    }

    public void midTraverse(){
        if(root == null){
            System.out.println("树的根结点为空！");
            return;
        }
        root.midTraverse();
    }

    public void delNode(int val){
        if(root == null){
            return;
        }
        root.delNode(val);
    }
    public void rotate(){
        if(root == null){
            return;
        }
        root.rotate();
    }
}
